1function [alpha_k, Kp, e, Ki] = fcn(p, WP)
 2
 3persistent index;
 4
 5if isempty(index)
 6     index = 2;
 7end
 8
 9
10
11
12%% THE LOOKAHEAD DISTANCE R
13R = 100;
14%%
15% The craft's position in NED
16x = p(1);   %[m] 
17y = p(2);   %[m]    
18
19
20
21% The next and previous waypoint in NED
22xk = WP(1, index - 1);                  xk_1 = WP(1, index); 
23yk = WP(2, index - 1);                  yk_1 = WP(2, index);
24
25% How far from the desired waypoint the craft must be before going to next
26% waypoint
27turning_Radius = 600; 
28
29if (((xk_1 - x)^2 + (yk_1 - y)^2) <= turning_Radius^2) && (length(WP) > index)  
30    index = index + 1;
31end
32
33
34
35%The angle between the North and two waypoints --> atan2(dy/dx)
36alpha_k =  atan2(yk_1 - yk, xk_1 - xk); %[rad] 
37
38%The rotation from the NED frame to path-fixed reference frame
39Rp = [  cos(alpha_k);   -sin(alpha_k);
40        sin(alpha_k);   cos(alpha_k);];
41    
42%How far the craft is from the previous way point
43p_tilde = [x,y]' - [xk,yk]'; % or p-WP_previous
44
45% The coordinates of the craft in the path-fixed reference frame (Fossen
46% p258)
47Epsilon = Rp * p_tilde;
48
49% Along track distance & cross track error
50s = Epsilon(1,:);
51e = Epsilon(2,:);
52
53%The look ahead distance DELTA
54DELTA = sqrt(R^2-e^2);
55
56% Steering law
57Kp = 1/DELTA;
58Ki = 0;
59end
